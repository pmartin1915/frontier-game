#### Audio Memory Leak Fix (MED-03)
```typescript
class AmbianceSystem {
  private static readonly MEMORY_LIMIT = 200 * 1024 * 1024; // 200MB
  private _tracks: Map<string, {track: Howl, lastUsed: number, size: number}> = new Map();
  private _totalMemory: number = 0;

  constructor() {
    setInterval(() => this.cleanupTracks(), 30000);
  }

  private cleanupTracks(): void {
    const now = Date.now();
    this._tracks.forEach((value, key) => {
      if (now - value.lastUsed > 180000) { // 3 minutes
        this._totalMemory -= value.size;
        value.track.unload();
        this._tracks.delete(key);
      }
    });

    if (this._totalMemory > AmbianceSystem.MEMORY_LIMIT * 0.9) {
      this.emergencyCleanup();
    }
  }

  private emergencyCleanup(): void {
    const sortedTracks = Array.from(this._tracks.entries())
      .sort((a, b) => a[1].lastUsed - b[1].lastUsed);
    while (this._totalMemory > AmbianceSystem.MEMORY_LIMIT * 0.8 && sortedTracks.length) {
      const [key, value] = sortedTracks.shift()!;
      this._totalMemory -= value.size;
      value.track.unload();
      this._tracks.delete(key);
    }
  }
}
```

#### River Crossing Fix (HIGH-03)
```typescript
interface JourneyState {
  riverCrossingInProgress: boolean;
  riverCrossingProgress: number; // 0-100%
  riverCrossingDifficulty: number; // 1-5
  riverCrossingStartTime: number; // Timestamp
}

class MovementSystem {
  calculateDistance(): number {
    if (this.gameState.journey.riverCrossingInProgress) {
      const elapsed = Date.now() - this.gameState.journey.riverCrossingStartTime;
      const progress = (elapsed / (10000 * this.gameState.journey.riverCrossingDifficulty)) * 100;
      this.gameState.journey.riverCrossingProgress = Math.min(progress, 100);

      if (this.gameState.journey.riverCrossingProgress >= 100) {
        this.gameState.journey.riverCrossingInProgress = false;
        return 0; // Crossing complete
      }
      return 0; // Still crossing
    }
    return 1; // Default distance
  }
}
```
