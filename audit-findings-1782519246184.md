### 4.2 Audio Memory Leak Fix (MED-03)
```typescript
class AmbianceSystem {
  private static readonly TRACK_TTL_MS = 300000; // 5 minutes
  private _tracks: Map<string, {track: Howl, lastUsed: number}> = new Map();

  constructor() {
    setInterval(() => this.cleanupTracks(), 60000);
  }

  private cleanupTracks(): void {
    const now = Date.now();
    this._tracks.forEach((value, key) => {
      if (now - value.lastUsed > AmbianceSystem.TRACK_TTL_MS) {
        value.track.unload();
        this._tracks.delete(key);
      }
    });
  }

  public destroy(): void {
    this._tracks.forEach(value => value.track.unload());
    this._tracks.clear();
  }
}
```
