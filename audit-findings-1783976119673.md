```typescript
class AmbianceSystem {
  private static readonly MEMORY_LIMIT = 200 * 1024 * 1024;
  private _tracks: Map<string, {track: Howl, lastUsed: number, size: number}> = new Map();
  private _totalMemory: number = 0;

  constructor() {
    setInterval(() => this.cleanupTracks(), 15000); // Increased frequency
  }

  private cleanupTracks(): void {
    const now = Date.now();
    this._tracks.forEach((value, key) => {
      if (now - value.lastUsed > 90000) { // Reduced from 3 minutes
        this._totalMemory -= value.size;
        value.track.unload();
        this._tracks.delete(key);
      }
    });

    if (this._totalMemory > AmbianceSystem.MEMORY_LIMIT * 0.8) {
      this.emergencyCleanup();
    }
  }
}
```
