#### River Crossing Fix (HIGH-03)
The following code snippet addresses the river crossing terrain lock by introducing a progress-based system, preventing the movement system from being compromised during a crossing.

```typescript
interface JourneyState {
  riverCrossingInProgress: boolean;
  riverCrossingProgress: number; // 0-100%
  riverCrossingDifficulty: number; // 1-5
}

class MovementSystem {
  // Assuming gameState is accessible, e.g., via constructor or context
  private gameState: { journey: JourneyState }; 

  constructor(gameState: { journey: JourneyState }) {
    this.gameState = gameState;
  }

  calculateDistance(): number {
    if (this.gameState.journey.riverCrossingInProgress) {
      const progress = 10 / this.gameState.journey.riverCrossingDifficulty;
      this.gameState.journey.riverCrossingProgress += progress;
      if (this.gameState.journey.riverCrossingProgress >= 100) {
        this.gameState.journey.riverCrossingInProgress = false;
        return 0; // Crossing complete, no distance moved in this step
      }
      return 0; // Still crossing, no distance moved in this step
    }
    return 1; // Default distance moved
  }
}
```
