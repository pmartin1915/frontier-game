class AmbianceSystem {
  private static readonly TRACK_TTL_MS = 180000; // Reduced from 5 to 3 minutes
  private static readonly MEMORY_LIMIT = 200 * 1024 * 1024; // 200MB hard limit
  private _tracks: Map<string, {track: Howl, lastUsed: number, size: number}> = new Map();
  private _totalMemory: number = 0;
  private _cleanupInterval: NodeJS.Timeout;

  constructor() {
    this._cleanupInterval = setInterval(() => this.cleanupTracks(), 30000);
    setInterval(() => this.logMemoryUsage(), 60000);
  }

  private cleanupTracks(): void {
    const now = Date.now();
    this._tracks.forEach((value, key) => {
      if (now - value.lastUsed > AmbianceSystem.TRACK_TTL_MS) {
        this._totalMemory -= value.size;
        value.track.unload();
        this._tracks.delete(key);
      }
    });

    if (this._totalMemory > AmbianceSystem.MEMORY_LIMIT * 0.9) {
      this.emergencyCleanup();
    }
  }

  private emergencyCleanup(): void {
    const sorted = Array.from(this._tracks.entries())
      .sort((a, b) => a[1].lastUsed - b[1].lastUsed);

    while (this._totalMemory > AmbianceSystem.MEMORY_LIMIT * 0.8 && sorted.length > 0) {
      const [key, value] = sorted.shift()!;
      this._totalMemory -= value.size;
      value.track.unload();
      this._tracks.delete(key);
    }
  }

  public destroy(): void {
    clearInterval(this._cleanupInterval);
    this._tracks.forEach(value => value.track.unload());
    this._tracks.clear();
    this._totalMemory = 0;
  }
}
